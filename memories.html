<script>


if(localStorage.getItem("loveAuth")!=="true"){

window.location.href="index.html";

}


let currentFilter="all";


let currentLang=

localStorage.getItem("lang") || "en";


const translations={


en:{

title:"📸 Our Memories",

subtitle:"Every beautiful moment together lives here ❤️",

dropTitle:"Drag & Drop Photos or Videos Here",

dropText:"Or use the Add Memory button",

all:"All",

photos:"Photos",

videos:"Videos",

special:"Special Moments",

add:"➕ Add Memory",

modal:"Add Memory"

},


ar:{

title:"📸 ذكرياتنا",

subtitle:"كل لحظة جميلة بيننا تعيش هنا ❤️",

dropTitle:"اسحب وأفلت الصور أو الفيديوهات هنا",

dropText:"أو استخدم زر إضافة ذكرى",

all:"الكل",

photos:"الصور",

videos:"الفيديوهات",

special:"اللحظات الخاصة",

add:"➕ إضافة ذكرى",

modal:"إضافة ذكرى"

}


};


function applyLanguage(){


const t=translations[currentLang];


document.documentElement.lang=currentLang;

document.documentElement.dir=

currentLang==="ar"?"rtl":"ltr";


document.getElementById("pageTitle").innerText=t.title;

document.getElementById("pageSubtitle").innerText=t.subtitle;


document.getElementById("dropTitle").innerText=t.dropTitle;

document.getElementById("dropText").innerText=t.dropText;


document.getElementById("filterAll").innerText=t.all;

document.getElementById("filterPhotos").innerText=t.photos;

document.getElementById("filterVideos").innerText=t.videos;

document.getElementById("filterSpecial").innerText=t.special;


document.getElementById("addMemoryBtn").innerText=t.add;

document.getElementById("modalTitle").innerText=t.modal;

}


function toggleLanguage(){


currentLang=

currentLang==="en"?"ar":"en";


localStorage.setItem("lang",currentLang);


applyLanguage();

}


applyLanguage();


function getMemories(){


return JSON.parse(

localStorage.getItem("loveMemories") || "[]"

);


}


function saveMemories(memories){


localStorage.setItem(

"loveMemories",

JSON.stringify(memories)

);


}


function openModal(){


document.getElementById("memoryModal").style.display="flex";


}


function closeModal(){


document.getElementById("memoryModal").style.display="none";


}


function saveMemory(){


const date=

document.getElementById("memoryDate").value;


const title=

document.getElementById("memoryTitle").value;


const desc=

document.getElementById("memoryDescription").value;


const mood=

document.getElementById("memoryMood").value;


const category=

document.getElementById("memoryCategory").value;


const file=

document.getElementById("memoryFile").files[0];


if(!title || !file){


alert("Please add title and file");


return;

}


const reader=new FileReader();


reader.onload=function(e){


const memories=getMemories();


memories.unshift({


id:Date.now(),


date:date,


title:title,


description:desc,


mood:mood,


category:category,


fileType:file.type,


fileData:e.target.result


});


saveMemories(memories);


renderGallery();


closeModal();


document.getElementById("memoryDate").value="";

document.getElementById("memoryTitle").value="";

document.getElementById("memoryDescription").value="";

document.getElementById("memoryFile").value="";


};


reader.readAsDataURL(file);


}


function deleteMemory(id){


let memories=getMemories();


memories=

memories.filter(m=>m.id!==id);


saveMemories(memories);


renderGallery();


}


function setFilter(filter){


currentFilter=filter;


renderGallery();


}


function renderGallery(){


const gallery=

document.getElementById("gallery");


gallery.innerHTML="";


let memories=getMemories();


if(currentFilter!=="all"){


memories=

memories.filter(

m=>m.category===currentFilter

);


}


if(memories.length===0){


gallery.innerHTML=`

<div style="

grid-column:1/-1;

text-align:center;

color:white;

font-size:20px;

padding:40px;

">

No memories yet ❤️

</div>

`;


return;

}


memories.forEach(memory=>{


const card=document.createElement("div");


card.className="memory-card";


let media="";


if(memory.fileType.startsWith("video")){


media=`

<video

src="${memory.fileData}"

onclick="openLightbox(

'${memory.fileData}',

'video'

)">

</video>

`;


}else{


media=`

<img

src="${memory.fileData}"

onclick="openLightbox(

'${memory.fileData}',

'image'

)">

`;

}


card.innerHTML=`


${media}


<div class="memory-content">


<div class="memory-title">

${memory.title}

</div>


<div class="memory-date">

${memory.date || ""}

</div>


<div class="memory-desc">

${memory.description || ""}

</div>


<div class="memory-mood">

${memory.mood}

</div>


<button

class="delete-btn"

onclick="deleteMemory(${memory.id})">


Delete


</button>


</div>


`;


gallery.appendChild(card);


});


}


function openLightbox(src,type){


const box=

document.getElementById("lightbox");


const content=

document.getElementById("lightboxContent");


if(type==="video"){


content.innerHTML=`

<video

src="${src}"

controls

autoplay>

</video>

`;


}else{


content.innerHTML=`

<img src="${src}">

`;


}


box.style.display="flex";


}


function closeLightbox(){


document.getElementById("lightbox")

.style.display="none";


}


const dropZone=

document.getElementById("dropZone");


dropZone.addEventListener(

"dragover",

e=>{


e.preventDefault();


dropZone.classList.add("dragover");


}

);


dropZone.addEventListener(

"dragleave",

()=>{


dropZone.classList.remove("dragover");


}

);


dropZone.addEventListener(

"drop",

e=>{


e.preventDefault();


dropZone.classList.remove("dragover");


const files=e.dataTransfer.files;


if(!files.length) return;


Array.from(files).forEach(file=>{


const reader=new FileReader();


reader.onload=function(ev){


const memories=getMemories();


memories.unshift({


id:Date.now()+Math.random(),


date:new Date()

.toISOString()

.split("T")[0],


title:file.name,


description:"",


mood:"❤️",


category:

file.type.startsWith("video")

?"video"

:"photo",


fileType:file.type,


fileData:ev.target.result


});


saveMemories(memories);


renderGallery();


};


reader.readAsDataURL(file);


});


}

);


window.onclick=function(e){


const modal=

document.getElementById("memoryModal");


if(e.target===modal){


closeModal();


}


};


renderGallery();
