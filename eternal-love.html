<script>


if(localStorage.getItem("loveAuth")!=="true"){

window.location.href="index.html";

}


let currentLang =

localStorage.getItem("lang") || "en";


const translations = {


en:{

title:"💌 Eternal Love",

subtitle:"Letters, promises and dreams forever.",

letters:"💌 Love Letters",

promises:"🤝 Future Promises",

bucket:"🌍 Shared Bucket List",

quote:"✨ Romantic Quote Of The Day"

},


ar:{

title:"💌 الحب الأبدي",

subtitle:"رسائل ووعود وأحلام إلى الأبد.",

letters:"💌 رسائل الحب",

promises:"🤝 وعود المستقبل",

bucket:"🌍 قائمة أحلامنا",

quote:"✨ اقتباس الحب اليومي"

}


};


function applyLanguage(){


const t = translations[currentLang];


document.documentElement.lang=currentLang;

document.documentElement.dir=

currentLang==="ar" ? "rtl":"ltr";


document.getElementById("pageTitle").innerText=t.title;

document.getElementById("pageSubtitle").innerText=t.subtitle;

document.getElementById("lettersTitle").innerText=t.letters;

document.getElementById("promisesTitle").innerText=t.promises;

document.getElementById("bucketTitle").innerText=t.bucket;

document.getElementById("quoteTitle").innerText=t.quote;


}


function toggleLanguage(){


currentLang =

currentLang==="en" ? "ar" : "en";


localStorage.setItem("lang",currentLang);


applyLanguage();


}


applyLanguage();


function getData(key){


return JSON.parse(

localStorage.getItem(key) || "[]"

);


}


function saveData(key,data){


localStorage.setItem(

key,

JSON.stringify(data)

);


}


/* =====================

LOVE LETTERS

===================== */


function addLetter(){


const author =

document.getElementById("letterAuthor").value.trim();


const text =

document.getElementById("letterText").value.trim();


if(!author || !text) return;


const letters=getData("loveLetters");


letters.unshift({


id:Date.now(),

author,

text,

date:new Date().toLocaleDateString()


});


saveData("loveLetters",letters);


document.getElementById("letterAuthor").value="";

document.getElementById("letterText").value="";


renderLetters();


}


function deleteLetter(id){


let letters=getData("loveLetters");


letters=

letters.filter(item=>item.id!==id);


saveData("loveLetters",letters);


renderLetters();


}


function renderLetters(){


const container=

document.getElementById("lettersList");


container.innerHTML="";


const letters=

getData("loveLetters");


letters.forEach(letter=>{


container.innerHTML += `


<div class="list-item">


<div class="item-text">


<strong>${letter.author}</strong>

<br><br>


${letter.text}


<br><br>


<small>${letter.date}</small>


</div>


<button

class="delete-btn"

onclick="deleteLetter(${letter.id})">


Delete


</button>


</div>


`;


});


}

function addPromise(){


const text=

document.getElementById("promiseInput").value.trim();


if(!text) return;


const promises=

getData("futurePromises");


promises.unshift({


id:Date.now(),

text:text


});

"futurePromises",

promises

);


document.getElementById("promiseInput").value="";


renderPromises();


}


function editPromise(id){


let promises=

getData("futurePromises");


const item=

promises.find(x=>x.id===id);


const newText=

prompt("Edit Promise",item.text);


if(!newText) return;


item.text=newText;


saveData(

"futurePromises",

promises

);


renderPromises();


}


function deletePromise(id){

getData("futurePromises");


promises=

promises.filter(

x=>x.id!==id

);

"futurePromises",

promises

}
